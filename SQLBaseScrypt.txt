   `age` int(11) NOT NULL DEFAULT '0',
   `isAdmin` bit(1) NOT NULL DEFAULT b'0',
   `date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
   PRIMARY KEY (`id`),
  KEY `date` (`createDate`)
 ) ENGINE=InnoDB DEFAULT CHARSET=utf8;
